% Add once to the paper preamble (after \documentclass), e.g.:
%   \input{nova_zenodo_doi_placeholder}
% or with a path if this file lives with your suite preamble:
%   \input{path/to/nova_zenodo_doi_placeholder}
%
% Then run: zpo doi inject --artifact <publication-id> --apply
% With --require-tex-markers, injection fails until this file is \input{} (markers must exist in the resolved .tex).
%
% Placement (SPEC / docs: DOI_PROPAGATION.md):
%   • Primary (recommended for all program papers): \NovaZenodoAfterTitle immediately after \maketitle.
%   • Optional duplicate: \NovaZenodoInAbstract before \end{abstract} (--abstract-doi-line on inject).
%
% The DOI value is written between % NOVA_ZPO_DOI_BEGIN / END by zpo doi inject (\def\NovaZenodoDOI{...}).
%
\makeatletter
% Expand #1 once, then test emptiness (plain \detokenize{\Macro} does NOT expand \Macro first).
\providecommand{\NovaZenodoIfDoiEmpty}[1]{%
  \if\relax\detokenize{#1}\relax
    \expandafter\@firstoftwo
  \else
    \expandafter\@secondoftwo
  \fi}
\makeatother
%
\providecommand{\NovaZenodoArchivalLine}{%
  \expandafter\NovaZenodoIfDoiEmpty\expandafter{\NovaZenodoDOI}%
  {\textit{Archival version (Zenodo):} \emph{[DOI pending publication]}}%
  {\textit{Archival version (Zenodo):} %
  \begingroup
  \edef\NovaZenodo@href{\endgroup\noexpand\href{https://doi.org/\NovaZenodoDOI}{https://doi.org/\NovaZenodoDOI}}%
  \NovaZenodo@href}%
}
%
% After \maketitle — canonical visible placement (first page, under title block).
\providecommand{\NovaZenodoAfterTitle}{%
  \par\smallskip
  \begingroup
  \centering
  \small
  \NovaZenodoArchivalLine
  \par
  \endgroup
  \smallskip
}
%
% End of abstract — optional; use when --abstract-doi-line is passed to zpo doi inject.
\providecommand{\NovaZenodoInAbstract}{%
  \par\smallskip\noindent
  \NovaZenodoArchivalLine
}
